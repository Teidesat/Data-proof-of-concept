\def\Register((#1,#2),#3) {
	\draw (#1, #2) -- (#1 + #3, #2) -- (#1 + #3, #2 + #3) -- (#1, #2 + #3) -- (#1, #2);
	\node[scale={1.5 * #3}] at (#1 + #3 / 2, #2 + #3 / 2) {D};
}

\def\Xor((#1, #2), #3) {
	\draw (#1, #2) circle (#3);
	\draw (#1 + #3, #2     ) -- (#1 - #3, #2);
	\draw (#1     , #2 + #3) -- (#1     , #2 - #3);
}

\section{RSC Structure}
\begin{tikzpicture}

\def\r{0.3};
\def\s{1};
\def\memorySize{3};
\def\sqSize{1};

\draw[->] (0,0) -- ({\s},0);
\Xor((1 + \r, 0), \r)

\foreach \i in {0,...,\memorySize}
{
	\def\step{1 + \r * 2 + \s * \i + \sqSize * \i};
	\def\y{\sqSize / -2};

	\draw[->] (\step, 0) -- ({\s + \step},0);
	\Register((\step + 1, \y), \sqSize)
}


\end{tikzpicture}
